{{/*
oms web-service:
  Template to run modelName_mpi.exe on Windows using Microsoft MPI

Rename this file into:
  mpi.ModelRun.template.txt
to use it by default to run models on MPI cluster.

Oms web-service process this template as:
  - skip empty lines
  - use first non-empty line as executable to run
  - substitute template arguments
  - combine result into single line
after template processing oms run it as: exec.Cmd.Start()

Arguments of template:
struct {
  ModelName string            // model name
  ExeStem   string            // base part of model exe name, usually modelName
  Dir       string            // work directory to run the model
  BinDir    string            // bin directory where model exe is located
  MpiNp     int               // number of MPI processes
  Args      []string          // model command line arguments
  Env       map[string]string // environment variables to run the model
}

Example of result:
  mpiexec -n 16 -wdir models\bin -env key value ..\bin\modelName_mpi -OpenM.LogToFile true

*/}}


mpiexec
-n
{{ .MpiNp}}
{{with .Dir}}
-wdir
{{.}}
{{end}}
{{range $key, $val := .Env}}
-env
{{$key}}
{{$val}}
{{end}}
{{.BinDir}}\{{.ExeStem}}_mpi
{{range .Args}}
{{.}}
{{end}}
